```javascript
const express = require('express');
const cors = require('cors');
const Anthropic = require('@anthropic-ai/sdk');
require('dotenv').config();

const app = express();
const client = new Anthropic({
  apiKey: process.env.ANTHROPIC_API_KEY
});

// Middleware
app.use(cors());
app.use(express.json());

// Rate limiting store (in production, use Redis)
const rateLimitStore = {};

// Tier limits
const TIER_LIMITS = {
  free: 5,
  mid: 500,
  premium: Infinity
};

// Helper: Get user's monthly usage
function getUserUsage(userId, tier) {
  const key = `${userId}:${new Date().toISOString().split('T')[0]}`;
  const current = rateLimitStore[key] || 0;
  return current;
}

// Helper: Increment user's usage
function incrementUsage(userId) {
  const key = `${userId}:${new Date().toISOString().split('T')[0]}`;
  rateLimitStore[key] = (rateLimitStore[key] || 0) + 1;
}

// Helper: Check rate limit
function checkRateLimit(userId, tier) {
  const usage = getUserUsage(userId, tier);
  const limit = TIER_LIMITS[tier];
  
  if (usage >= limit) {
    return {
      allowed: false,
      message: \`Rate limit exceeded. \${tier} tier: \${usage}/\${limit} analyses used today.\`
    };
  }
  
  return {
    allowed: true,
    remaining: limit - usage - 1
  };
}

// Analysis prompts
const PROMPTS = {
  scanner: (text) => \`You are an expert marketing psychologist and behavioral economist. You analyze marketing copy for psychological triggers and emotional hooks. You understand the 8 key emotional triggers: Urgency, Scarcity, Social Proof, Authority, Fear, Value, Identity, Curiosity.

Analyze this copy for emotional triggers. Return a structured analysis:

1. TRIGGERS DETECTED: List which of the 8 triggers are present with impact scores (0-10)
2. STRONGEST HOOK: Identify the most powerful emotional trigger
3. MISSING TRIGGERS: Which triggers could strengthen the copy?
4. IMPROVEMENT: Suggest specific copy changes with the triggers they activate
5. EFFECTIVENESS SCORE: Overall persuasiveness (0-100%)
6. RECOMMENDATION: One key change that would increase conversion rate

Format: Clear sections with bullet points. Be specific and actionable.\`,

  rewriter: (text) => \`You are an expert marketing psychologist and behavioral economist. You analyze marketing copy for psychological triggers and emotional hooks. You understand the 8 key emotional triggers: Urgency, Scarcity, Social Proof, Authority, Fear, Value, Identity, Curiosity.

Generate 6 variations of this copy, each optimized for a different buyer archetype:

For each, provide the rewritten copy (2-3 sentences). Make them distinctly different.\`,

  strategist: (text) => \`You are an expert marketing psychologist and behavioral economist. You analyze marketing copy for psychological triggers and emotional hooks. You understand the 8 key emotional triggers: Urgency, Scarcity, Social Proof, Authority, Fear, Value, Identity, Curiosity.

// POST /api/analyze - Main analysis endpoint
app.post('/api/analyze', async (req, res) => {
  try {
    const { type, text, userId, tier } = req.body;

    // Validation
    if (!type || !text || !userId || !tier) {
      return res.status(400).json({
        error: 'Missing required fields: type, text, userId, tier'
      });
    }

    if (!PROMPTS[type]) {
      return res.status(400).json({
        error: \`Invalid analysis type: \${type}\`
      });
    }

    if (!['free', 'mid', 'premium'].includes(tier)) {
      return res.status(400).json({
        error: \`Invalid tier: \${tier}\`
      });
    }

    // Check rate limit
    const limitCheck = checkRateLimit(userId, tier);
    if (!limitCheck.allowed) {
      return res.status(429).json({
        error: limitCheck.message
      });
    }

    // Call Claude API
    const message = await client.messages.create({
      model: 'claude-3-5-sonnet-20241022',
      max_tokens: 1024,
      messages: [
        {
          role: 'user',
          content: PROMPTS[type](text)
        }
      ]
    });

    // Extract response
    const analysisText = message.content[0].type === 'text' 
      ? message.content[0].text 
      : 'No response generated';

    // Increment usage
    incrementUsage(userId);

    // Return result
    res.json({
      success: true,
      analysis: analysisText,
      timestamp: new Date().toISOString(),
      usage: {
        current: getUserUsage(userId, tier) + 1,
        limit: TIER_LIMITS[tier],
        remaining: limitCheck.remaining
      }
    });

  } catch (error) {
    console.error('Analysis error:', error);

    // Handle specific API errors
    if (error.status === 401) {
      return res.status(401).json({
        error: 'API key invalid or expired'
      });
    }

    if (error.status === 429) {
      return res.status(429).json({
        error: 'Claude API rate limit exceeded. Try again in a moment.'
      });
    }

    if (error.status === 500) {
      return res.status(500).json({
        error: 'Claude API error. Please try again later.'
      });
    }

    res.status(500).json({
      error: error.message || 'Internal server error'
    });
  }
});

// GET /api/health - Health check
app.get('/api/health', (req, res) => {
  res.json({
    status: 'ok',
    timestamp: new Date().toISOString()
  });
});

// GET /api/usage/:userId/:tier - Get current usage
app.get('/api/usage/:userId/:tier', (req, res) => {
  const { userId, tier } = req.params;

  if (!['free', 'mid', 'premium'].includes(tier)) {
    return res.status(400).json({ error: 'Invalid tier' });
  }

  const usage = getUserUsage(userId, tier);
  const limit = TIER_LIMITS[tier];

  res.json({
    userId,
    tier,
    usage,
    limit,
    remaining: Math.max(0, limit - usage)
  });
});

// Error handling middleware
app.use((err, req, res, next) => {
  console.error('Unhandled error:', err);
  res.status(500).json({
    error: 'Internal server error'
  });
});

// Start server
const PORT = process.env.PORT || 3001;
app.listen(PORT, () => {
  console.log(\`✓ AI Analysis Server running on port \${PORT}\`);
  console.log(\`✓ Claude Model: claude-3-5-sonnet-20241022\`);
});
```

## SECURITY SUMMARY

### ✅ Critical Fixes Applied

1. **API Key Protection**
   - Moved from frontend to backend environment variables
   - Never exposed in client-side code
   - Transmitted securely via HTTPS

2. **Rate Limiting**
   - Free: 5 analyses/month
   - Creator: 500 analyses/month
   - Premium: Unlimited
   - Per-user tracking
   - Daily reset at midnight UTC

3. **Error Handling**
   - 400: Invalid request parameters
   - 401: API authentication failure
   - 429: Rate limit exceeded
   - 500: Server errors
   - User-friendly error messages

4. **Input Validation**
   - All parameters validated
   - Type checking on tier and analysis type
   - Text content validated before sending to Claude

5. **CORS Configuration**
   - Backend accepts requests from authorized origins
   - Prevents cross-origin attacks
